8/// A collection of issues.
9pub struct Issues {
10    list: Vec<Issue>,
11    labels: Vec<String>,
12}
13
14impl easy_storage::Storeable for Issues {}
15
16impl Issues {
17    /// Creates a new `Issues` collection.
18    ///
19    /// This initializes the collection with a "root" issue.
20    pub fn new() -> Self {
21        let now = chrono::Local::now();
22        let root_issue = Issue {
23            name: "root".to_string(),
24            status: Status::CloseAsCmp,
25            comment: Vec::new(),
26            created_by: User::new("root", "root"),
27            from: usize::MAX,
28            created_at: now,
29            updated_at: now,
30            labels: Vec::new(),
31        };
32        Self {
33            list: vec![root_issue],
34            labels: vec!["Bug".to_string(), "Enhance".to_string()],
35        }
36    }
37
38    /// Adds a new issue to the collection.
39    ///
40    /// Returns the index of the newly added issue.
41    pub fn add_new_issue(&mut self, i: Issue) -> usize {
42        let l = self.list.len();
43        self.list.push(i);
44        l
45    }
46
47    /// Returns a reference to the list of issues.
48    pub fn get_list(&self) -> &Vec<Issue> {
49        &self.list
50    }
51
52    /// Finds issues containing the given title string.
53    ///
54    /// Returns `Some(Vec<&mut Issue>)` if matches are found, otherwise `None`.
55    pub fn find_from_title<T: AsRef<str>>(&mut self, s: T) -> Option<Vec<&mut Issue>> {
56        let res = self
57            .list
58            .iter_mut()
59            .filter(|f| f.name.contains(s.as_ref()))
60            .collect::<Vec<_>>();
61        if res.is_empty() { None } else { Some(res) }
62    }
63
64    /// Finds issues updated within the given time range.
65    ///
66    /// Returns `Some(Vec<&mut Issue>)` if matches are found, otherwise `None`.
67    pub fn find_from_updated_time(
68        &mut self,
69        st: chrono::DateTime<chrono::Local>,
70        ed: chrono::DateTime<chrono::Local>,
71    ) -> Option<Vec<&mut Issue>> {
72        let res = self
73            .list
74            .iter_mut()
75            .filter(|f| (st..=ed).contains(&f.updated_at))
76            .collect::<Vec<_>>();
77        if res.is_empty() { None } else { Some(res) }
78    }
79
80    /// Finds issues created within the given time range.
81    ///
82    /// Returns `Some(Vec<&mut Issue>)` if matches are found, otherwise `None`.
83    pub fn find_from_created_time(
84        &mut self,
85        st: chrono::DateTime<chrono::Local>,
86        ed: chrono::DateTime<chrono::Local>,
87    ) -> Option<Vec<&mut Issue>> {
88        let res = self
89            .list
90            .iter_mut()
91            .filter(|f| (st..=ed).contains(&f.created_at))
92            .collect::<Vec<_>>();
93        if res.is_empty() { None } else { Some(res) }
94    }
95
96    /// Finds issues containing a comment with the given text.
97    ///
98    /// Returns `Some(Vec<&mut Issue>)` if matches are found, otherwise `None`.
99    pub fn find_from_comments<T: AsRef<str>>(&mut self, s: T) -> Option<Vec<&mut Issue>> {
100        let res = self
101            .list
102            .iter_mut()
103            .filter(|f| f.contains_comment(s.as_ref()))
104            .collect::<Vec<_>>();
105        if res.is_empty() { None } else { Some(res) }
106    }
107
108    /// Gets an issue by its index.
109    pub fn get(&self, index: usize) -> Option<&Issue> {
110        self.list.get(index)
111    }
112
113    /// Gets a mutable reference to an issue by its index.
114    pub fn get_mut(&mut self, index: usize) -> Option<&mut Issue> {
115        self.list.get_mut(index)
116    }
117
118    /// Forks an issue from a given index.
119    ///
120    /// The original issue is marked as `CloseAsForked`, and a new copy is created
121    /// with `from` set to the original index.
122    ///
123    /// Returns `Some(usize)` which is the index of the new forked issue, or `None` if the original issue doesn't exist.
124    pub fn fork(&mut self, from: usize) -> Option<usize> {
125        let mut forked = self.get(from)?.clone();
126        forked.from = from;
127        self.get_mut(from).unwrap().status = Status::CloseAsForked;
128        Some(self.add_new_issue(forked))
129    }
130
131    /// Returns a list of all registered labels.
132    pub fn get_registered_labels(&self) -> Vec<String> {
133        self.labels.clone()
134    }
135}

137#[derive(Debug, Clone, Serialize, Deserialize)]
138/// Represents a single issue.
139pub struct Issue {
140    name: String,
141    status: Status,
142    comment: Vec<Comment>,
143    created_by: User,
144    created_at: chrono::DateTime<chrono::Local>,
145    updated_at: chrono::DateTime<chrono::Local>,
146    labels: Vec<String>,
147    from: usize,
148}
149
150impl Issue {
151    /// Creates a new issue with a name and the user who created it.
152    pub fn new<T: AsRef<str>, U: AsRef<str>>(name: T, user: User, labels: Vec<U>) -> Self {
153        let now = chrono::Local::now();
154        let labels = labels
155            .iter()
156            .map(|f| f.as_ref().to_string())
157            .collect::<Vec<_>>();
158        Self {
159            name: name.as_ref().to_string(),
160            status: Status::default(),
161            comment: Vec::new(),
162            created_by: user,
163            from: 0,
164            created_at: now,
165            updated_at: now,
166            labels,
167        }
168    }
169
170    /// Returns a list of labels associated with this issue.
171    pub fn get_labels(&self) -> Vec<String> {
172        self.labels.clone()
173    }
174
175    /// Updates the `updated_at` timestamp to the current local time.
176    pub fn update(&mut self) {
177        self.updated_at = chrono::Local::now();
178    }
179
180    /// Closes the issue as completed.
181    pub fn close_as_cmp(&mut self) {
182        self.status = Status::CloseAsCmp;
183        self.update();
184    }
185
186    /// Closes the issue as not planned.
187    pub fn close_as_not_planed(&mut self) {
188        self.status = Status::CloseAsNotPlaned;
189        self.update();
190    }
191
192    /// Closes the issue as forked.
193    pub fn close_as_forked(&mut self) {
194        self.status = Status::CloseAsForked;
195        self.update();
196    }
197
198    /// Adds a comment entry to the issue.
199    pub fn comment(&mut self, new_comment: Comment) {
200        self.comment.push(new_comment);
201        self.update();
202    }
203
204    /// Checks if any comment in the issue contains the given string.
205    pub fn contains_comment<T: AsRef<str>>(&self, s: T) -> bool {
206        let res = self
207            .comment
208            .iter()
209            .filter(|f| f.content.contains(s.as_ref()))
210            .collect::<Vec<_>>();
211        !res.is_empty()
212    }
213}

216/// Represents the status of an issue.
217pub enum Status {
218    #[default]
219    /// The issue is open.
220    Open,
221    /// The issue is closed as completed.
222    CloseAsCmp,
223    /// The issue is closed as not planned.
224    CloseAsNotPlaned,
225    /// The issue is closed because it was forked.
226    CloseAsForked,
227}
